162  pointer allocate (size_type n) {
163  // can't allocate negative sized blocks, throw bad_alloc
164  if (n < 0) {
165  throw std::bad_alloc();
166  }
167  if (n == 0) { //doesn't allocate space and returns null
168  return nullptr;
169  }
170  int i = 0;
171  bool not_done = true;
172  while (i < N && not_done) {
173  int s = (*this)[i];
174  if (s < 0) { //skips past allocated block
175  i += -s + (2 * sizeof(int));
176  }
177  else if (s < (n * sizeof(T))) { //skips past blocks that are too small
178  i += s + (2 * sizeof(int));
179  }
180  else { //acually allocates space
181  if (s - ( n * sizeof(T) ) < (2 * sizeof(int) + sizeof(T))) //coalesces blocks that are too small
182  {
183  (*this)[i] = -s;
184  (*this)[i + s + sizeof(int)] = -s;
185  }
186  else if (s - n == 0) //simply adapts all the space at the sentinel
187  {
188  (*this)[i] = -s;
189  (*this)[i + s + sizeof(int)] = -s;
190  }
191  else { //allocates space and changes sentinels to match the new space remaining
192  (*this)[i] = -n * sizeof(T);
193  (*this)[i + n * sizeof(T) + sizeof(int)] = -n * sizeof(T);
194  (*this)[i + n * sizeof(T) + 2 * sizeof(int)] = s - n * sizeof(T) - (2 * sizeof(int));
195  (*this)[i + s + sizeof(int)] = s - n * sizeof(T) - (2 * sizeof(int));
196  }
197 
198  not_done = false;
199  }
200  }
201  if (i == N) //this means there wasn't enough room for allocation
202  {
203  throw std::bad_alloc();
204  }
205 
206  assert(valid());
207  return reinterpret_cast<T*>(&a[i + sizeof(int)]);
208  }
